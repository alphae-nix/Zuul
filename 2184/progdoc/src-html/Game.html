054    /**
055     * Va afficher un message de bienvenu au début du jeu
056     */
057    private void printWelcome() {
058        System.out.println("Welcome to the World of Zuul! ") ;
059        System.out.println("You must find a combination of numbers corresponding to the geographical coordinates of a rebel base that is fighting against the \"party\"");

066
067    /**
068     * Va nous permettre de lancer le jeu 
069     */
070    public void play() {
071        printWelcome() ;
072        boolean vFinished = false ;
073        while( vFinished == false){
074            Command vCommand = this.aParser.getCommand();
075            vFinished = this.processCommand( vCommand ) ;
076        }
077        System.out.println( "Thank you for playing. Good bye." );
078
079    } //play() 
080
081    /**
082     * Va permettre d'initialiser les attributs de la classe
083     */
084    public Game () {
085        this.createRooms() ;
086        aParser = new Parser();
087    } // Constructeur par Défaut
088
089    /**
090     * Procédure goHome exécuté quand on tape la commande "go" et qui permet de changer de pièce
091     * @param pCommand la commande passé en paramètre pour connaitre la direction
092     */
093    private void goRoom (final Command pCommand)
094    { 
095        if ( ! pCommand.hasSecondWord() ){
096            System.out.println("Go where");
097            return ;
098        }//Vérification de l'insertion d'un deuxième caractère et affichage message d'erreur
099
100        String vDirection = pCommand.getSecondWord() ;
101
102        Room vNextRoom = aCurrentRoom.getExit(vDirection) ; // initialisation de NextRoom à null
103
104        if (vNextRoom==null){
105            System.out.println("There is no door !") ;
106        }
107
108        else{
109            this.aCurrentRoom=vNextRoom ;
110            /*System.out.println(vNextRoom.getDescription()) ;*/
111            printLocationInfo();
112        }
113    } //fin goRoom
114
115    /**
116     * Va afficher un message lorsqu'on tapera "help"
117     */
118    private void printHelp(){
119        System.out.println("You are lost. You are alone.\n" +
120            " Your command words are:\n" + "   go quit help ") ;
121    } // fin méthode printHelp
122
123    /**
124     * Va initialiser la fonction Boolean qui se lance quand on marquera "quit"
125     * @param pCommand est la commande tapé au clavier
126     * @return la valeur de quit
127     */
128    private boolean quit(final Command pCommand){
129        if ( pCommand.hasSecondWord()){
130            System.out.println("Quit what ?") ;
131            return false ;
132        }
133        return true ; 
134
135    } // fin méthode quit
136
137    /**
138     * Va lister des commandes qui seront validées lorsqu'on les écrits dans le jeu
139     * @param pCommand commande tapé
140     * @return la valeure de processCommand
141     */
142    private boolean processCommand( final Command pCommand){
143        String vCommandWord = pCommand.getCommandWord() ;
144        if (pCommand.isUnknown()==true){
145            System.out.println("I don't know what you mean...") ;
146            return false ;
147        }
148
149        if (vCommandWord.equals("quit")){
150            return quit(pCommand) ;
151        }
152
153        if (vCommandWord.equals("go")){
154            goRoom(pCommand) ;
155            return false ;
156        }
157
158        if (vCommandWord.equals("help")){
159            printHelp() ;
160            return false ;
161        }
162
163        return false ;
164    }
165
166    /**
167     * Va permettre d'afficher les informations relatives au lieu où on se trouve
168     */
169    private void printLocationInfo(){
170        System.out.println(aCurrentRoom.getDescription());
171        System.out.print("List of exits :");
172        System.out.println(aCurrentRoom.getExitString());
173
174    }
