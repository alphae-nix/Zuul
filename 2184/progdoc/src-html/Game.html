004    private Room aCurrentRoom;

006    /**

008     */

010        Room vHome= new Room ("You are home") ;
011        Room vStreet1 = new Room ("You are in the First street") ;
012        Room vFactory = new Room ("You are in the party of truth") ;
013        Room vStreet2 = new Room ("You are in the Second Street") ;
014        Room vStreet3 = new Room ("You are in are in the Third Street") ;
015        Room vPark = new Room ("You are in Hyde Park") ;
016        Room vBridge = new Room ("You are in Tower Bridge");
017        Room vCafe = new Room ("You are in the Café du Chataigner");
018        Room vStreet4 = new Room ("You are in the Fourth Street") ;
019        Room vCinema = new Room ("You are in the Cinema");
020        Room vCave = new Room (" You are in a secret place") ;

022

024        vStreet1.setExits ( vFactory, null , vHome , null ) ;

040
041    /**
042     * Va afficher un message de bienvenu au début du jeu
043     */
044    private void printWelcome() {
045        System.out.println("Welcome to the World of Zuul! ") ;
046        System.out.println("You must find a combination of numbers corresponding to the geographical coordinates of a rebel base that is fighting against the \"party\"");
047        System.out.println("You will need to follow the instructions in your book to avoid being arrested.");
048
049        System.out.println("Type 'help' if you need help.");
050        System.out.println('\n');
051        printLocationInfo();
052    } //Affichage du message au début du jeu 
053    
054    /**
055     * Va nous permettre de lancer le jeu 
056     */
057    public void play() {
058        printWelcome() ;
059        boolean vFinished = false ;
060        while( vFinished == false){
061            Command vCommand = this.aParser.getCommand();
062            vFinished = this.processCommand( vCommand ) ;
063        }
064        System.out.println( "Thank you for playing. Good bye." );
065
066    } //play() 
067    
068    /**
069     * Va permettre d'initialiser les attributs de la classe
070     */
071    public Game () {
072        this.createRooms() ;
073        aParser = new Parser();
074    } // Constructeur par Défaut
075
076    /**
077     * Procédure goHome exécuté quand on tape la commande "go" et qui permet de changer de pièce
078     * @param pCommand la commande passé en paramètre pour connaitre la direction
079     */
080    private void goRoom (final Command pCommand)
081    { 
082        if ( ! pCommand.hasSecondWord() ){
083            System.out.println("Go where");
084            return ;
085        }//Vérification de l'insertion d'un deuxième caractère et affichage message d'erreur
086
087        String vDirection = pCommand.getSecondWord() ;
088
089        Room vNextRoom = aCurrentRoom.getExit(vDirection) ; // initialisation de NextRoom à null
090
091        if (vNextRoom==null){
092            System.out.println("There is no door !") ;
093        }
094
095        else{
096            this.aCurrentRoom=vNextRoom ;
097            /*System.out.println(vNextRoom.getDescription()) ;*/
098            printLocationInfo();
099        }
100    } //fin goRoom
101    
102    /**
103     * Va afficher un message lorsqu'on tapera "help"
104    */
105    private void printHelp(){
106        System.out.println("You are lost. You are alone.\n" +
107            " Your command words are:\n" + "   go quit help ") ;
108    } // fin méthode printHelp
109    
110    /**
111     * Va initialiser la fonction Boolean qui se lance quand on marquera "quit"
112     * @param pCommand est la commande tapé au clavier
113     * @return la valeur de quit
114     */
115    private boolean quit(final Command pCommand){
116        if ( pCommand.hasSecondWord()){
117            System.out.println("Quit what ?") ;
118            return false ;
119        }
120        return true ; 
121
122    } // fin méthode quit
123
124    /**
125     * Va listé des commandes qui seront validées lorsqu'on les écrits dans le jeu
126     * @param pCommand commande tapé
127     * @return la valeure de processCommand
128     */
129    private boolean processCommand( final Command pCommand){
130        String vCommandWord = pCommand.getCommandWord() ;
131        if (pCommand.isUnknown()==true){
132            System.out.println("I don't know what you mean...") ;
133            return false ;
134        }
135
136        if (vCommandWord.equals("quit")){
137            return quit(pCommand) ;
138        }
139
140        if (vCommandWord.equals("go")){
141            goRoom(pCommand) ;
142            return false ;
143        }
144
145        if (vCommandWord.equals("help")){
146            printHelp() ;
147            return false ;
148        }
149
150        return false ;
151    }
152

155     */

167        }
168        if(aCurrentRoom.getExit("west") != null){
169            System.out.print("west ") ;
170        }
171        System.out.println();
172    }
