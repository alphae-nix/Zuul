001import java.util.HashMap;
002import java.util.Set;
003/**
004 * Cette classe représente une salle, elle référence des liens vers des salles voisines
005 */
006public class Room
007{
008    private String aDescription;
009    private HashMap<String, Room> aExits;
010    // Attributs   
011    /**
012     * Va initialiser les attributs de la classe
013     * @param pDescription la nouvelle description de la room
014     */
015    public Room (final String pDescription){ //Modifier pour l'exercice 7.8
016        this.aDescription = pDescription ;
017        aExits = new HashMap<>();
018    } // Constructeur naturel
019
020    /**
021     * Va créer un accesseur de aDescription
022     * @return renvoi la description de l'objet 
023     */
024    public String getDescription() { //Modifier pour l'exercice 7.8
025        return this.aDescription;
026    } // Accesseur de Description
027
028    /**
029     * Définit les sorties de cette pièce. Chaque direction soit conduit à une autre pièce, soit est null (il n'y a pas de sortie dans cette direction)
030     * @param pDirection direction de la salle voisine à ajouter 
031     * @param pNeighbor salle voisine à ajouter
032     */
033    public void setExits (final String pDirection, final Room pNeighbor) { //Modifier pour l'exercice 7.8
034        aExits.put(pDirection, pNeighbor);
035    } //Procédure, hashmap
036
037    /**
038     * Cette méthode retourne la salle dans la direction donné en argument, si il n'y a pas de salle dans la direction donnée en argument alors renvoi null
039     * @param pDirection Chaine de charactère tapé par l'utilisateur qui va lui affecter une direction
040     * @return les différentes direction
041     */
042    public Room getExit(final String pDirection){ //Modifier pour l'exercice 7.8
043        return aExits.get(pDirection);
044    }
045    
046    /**
047     * Donne la liste des sortis possibles pour cette pièce
048     * @return renvoi la liste des sortis possibles
049     */
050    public String getExitString(){ //Modifié pour l'exercice 7.9
051        String listOfExits = "";
052        Set<String> exitsKeys = this.aExits.keySet();
053        for (String exitKey: exitsKeys){ // : signnifie pour chaques
054            listOfExits += exitKey + " ";
055        }
056        return listOfExits;
057    } //fin getExitString
058} // Room
