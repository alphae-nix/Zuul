003public class Room
004{
005    private String aDescription = "Cette salle est une chambre ou réside notre héro, elle n'est pas très bien entretenu et il n'a qu'une seule pièce" ;
006    public Room aNorthExit ;
007    public Room aEastExit ;
008    public Room aSouthExit ;
009    public Room aWestExit ;
010    // Attributs   
011 /**
012  * Va initialiser les attributs de la classe
013  * @param pDescription la nouvelle description de la room
014  */
015 public Room (final String pDescription){
016    this.aDescription = pDescription ;
017
018 } // Constructeur naturel
019 
020 /**
021  * Va créer un accesseur de aDescription
022  * @return renvoi la description de l'objet
023  */
024 public String getDescription() {
025    return this.aDescription;
026  } // Accesseur de Description
027 
028 /**
029  * Va initialiser les différentes sorties possibles pour une pièce
030  * @param pNorthExit initialise une sortie au Nord de la pièce
031  * @param pEastExit initialise une sortie à l'Est de la pièce
032  * @param pSouthExit initialise une sortie au Sud de la pièce
033  * @param pWestExit initialise une sortie à l'Ouest de la pièce
034  */
035 public void setExits (final Room pNorthExit , final Room pEastExit , final Room pSouthExit , final Room pWestExit) {
036    this.aNorthExit = pNorthExit ;
037    this.aEastExit = pEastExit ; 
038    this.aSouthExit = pSouthExit ; 
039    this.aWestExit = pWestExit ;
040  } //Procédure
041 
042  /**
043   * Permet de comparer pDirection avec les différentes directions initialisés
044   * @param pDirection Chaine de charachtère tapé par l'utilisateur qui va lui affecter une direction
045   * @return les différentes direction
046   */
047 public Room getExit(final String pDirection){
048     if (pDirection.equals("north")){
049         return aNorthExit;
050     }
051     if (pDirection.equals("east")){
052         return aEastExit;
053     }
054     if (pDirection.equals("south")){
055         return aSouthExit;
056     }
057     if (pDirection.equals("west")){
058         return aWestExit;
059     }
060     return null;
061 }
062} // Room
