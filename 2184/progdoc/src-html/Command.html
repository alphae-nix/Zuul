001/**
002 * Cette classe va représenter les commandes tapées par le joueur dans la console
003 */
004public class Command
005{
006    private String aCommandWord = null ;
007    private String aSecondWord = null ;
008    // Attributs
009
010    /**
011     * Va initialiser les différents attributs de la classe
012     * @param pCommandWord lis le premier mot
013     * @param pSecondWord lis le deuxième mot
014     */
015    public Command (final String pCommandWord , final String pSecondWord){
016        this.aCommandWord = pCommandWord ;
017        this.aSecondWord = pSecondWord ;
018    } // Constructeur naturel
019
020    /**
021     * Va créer un accesseur à l'attribut aCommandWord()
022     * @return aCommandWord
023     */
024    public String getCommandWord()
025    {
026        return this.aCommandWord ;
027    } // Accesseur CommandWord
028
029    /**
030     * Va créer un accesseur à l'attribut aSecondWord()
031     * @return aSecondWord
032     */
033    public String getSecondWord(){
034        return this.aSecondWord ;
035    } // Accesseur SecondWord
036
037    /**
038     * Vérification qu'un second mot a bien été tapé
039     * @return va retourner le deuxième mot
040     */
041    public boolean hasSecondWord(){
042        boolean hasSecondWord = false ;
043        if ( this.aSecondWord == null ){
044            hasSecondWord = false ;
045        }
046
047        else hasSecondWord = true ;
048        return hasSecondWord ;
049    } // fin hasSecondWord()
050
051    /**
052     * Va vérifier qu'une méthode à bien été tapé
053     * @return si la commande est connue ou pas
054     */
055    public boolean isUnknown() {
056        return (this.aCommandWord == null ) ;
057    } // fin isUknown()
058
059} // Command
