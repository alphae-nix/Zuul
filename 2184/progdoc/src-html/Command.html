002public class Command
003{
004    private String aCommandWord = null ;
005    private String aSecondWord = null ;
006    // Attributs
007
008    /**
009     * Va initialiser les différents attributs de la classe
010     * @param pCommandWord lis le premier mot
011     * @param pSecondWord lis le deuxième mot
012     */
013    public Command (final String pCommandWord , final String pSecondWord){
014        this.aCommandWord = pCommandWord ;
015        this.aSecondWord = pSecondWord ;
016    } // Constructeur naturel
017
018    /**
019     * Va créer un accesseur à l'attribut aCommandWord()
020     * @return aCommandWord
021     */
022    public String getCommandWord()
023    {
024        return this.aCommandWord ;
025    } // Accesseur CommandWord
026
027    /**
028     * Va créer un accesseur à l'attribut aSecondWord()
029     * @return aSecondWord
030     */
031    public String getSecondWord(){
032        return this.aSecondWord ;
033    } // Accesseur SecondWord
034
035    /**
036     * Vérification qu'un second mot a bien été tapé
037     * @return va retourner le deuxième mot
038     */
039    public boolean hasSecondWord(){
040        boolean hasSecondWord = false ;
041        if ( this.aSecondWord == null ){
042            hasSecondWord = false ;
043        }
044
045        else hasSecondWord = true ;
046        return hasSecondWord ;
047    } // fin hasSecondWord()
048
049    /**
050     * Va vérifier qu'une méthode à bien été tapé
051     * @return si la commande est connue ou pas
052     */
053    public boolean isUnknown() {
054        return (this.aCommandWord == null ) ;
055    } // fin isUknown()
056
057} // Command
